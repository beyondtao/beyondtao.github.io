package main

import (
    "io"
    "os"
)

func main() {
    // и®ҫе®ҡе·ҘдҪңи·Ҝеҫ„, дёҖиҲ¬еңЁйЎ№зӣ®еҲҡејҖе§Ӣи®ҫе®ҡдёҖж¬ЎеҚіеҸҜ, дёҚиҰҒеңЁејӮжӯҘиҝҮзЁӢдёӯдҝ®ж”№е·ҘдҪңи·Ҝеҫ„
    // й»ҳи®ӨжҳҜзЁӢеәҸжү§иЎҢзҡ„и·Ҝеҫ„
    os.Chdir("./")

    // еҲӣе»әж–Үд»¶еӨ№
    os.MkdirAll("./aa/bb/c1", 0777)	//os.ModePerm е°ұжҳҜ0777
    os.MkdirAll("./aa/bb/c2", 0777)

    // еҲӣе»әж–Үд»¶
    os.Create("./aa/bb/c1/file.go")

    // з§»еҠЁж–Үд»¶
    os.Rename("./aa/bb/c1/file.go", "./aa/bb/c2/file.go")

    // жү“ејҖж–Үд»¶,еҫ—еҲ°дёҖдёӘ *File еҜ№иұЎ, з”ЁдәҺеҗҺз»ӯзҡ„еҶҷе…Ҙ
    file, _ := os.OpenFile("./aa/bb/c2/file.go", 2, 0666)

    // еҶҷе…Ҙж–Үд»¶еҶ…е®№
    io.WriteString(file, `
        package main
        func main(){
            log.Println("жҲ‘жҳҜз”ұзЁӢеәҸеҶҷе…Ҙзҡ„д»Јз Ғ")
        }
    `)

    // д№ҹеҸҜд»ҘзӣҙжҺҘи°ғз”ЁfileйҮҢзҡ„еҮҪж•°еҶҷе…ҘеҶ…е®№
    file.WriteString("add string")

    // жӢ·иҙқж–Үд»¶, жӢ·иҙқе…¶е®һе°ұжҳҜеҲӣе»әдёҖдёӘж–Үд»¶, з„¶еҗҺеҶҷе…Ҙж–Үд»¶еҶ…е®№
    src1, _ := os.Create("./aa/bb/c1/file-copy1.go")
    io.Copy(file, src1) // жҠҠж–Үд»¶file, еҶҷе…Ҙsrc1ж–Үд»¶

    // еҲ йҷӨж–Үд»¶жҲ–ж–Үд»¶еӨ№
    os.Create("./aa/bb/c1/file-delete.go") // еҲӣе»әдёҖдёӘж–Үд»¶з”ЁдәҺеҲ йҷӨ
    os.RemoveAll("./aa/bb/c1/file-delete.go")

}

ж–Үд»¶иҰҶзӣ–еҲӨж–ӯ os.IsNotExist

еҪ“ж–Үд»¶е·ІеӯҳеңЁж—¶, дёҚз®ЎжҳҜos.Rename(), os.Create(), иҝҳжҳҜio.WriteString() йғҪдјҡеҜ№е·ІеӯҳеңЁзҡ„ж–Үд»¶иҝӣиЎҢиҰҶзӣ–\дҝ®ж”№,

package main

import (
    "io"
    "os"
)

func main() {
    os.MkdirAll("aa", 0777)

    // еҲӣе»әдәҶж–Үд»¶ testA, е№¶дё”еҶҷе…ҘдәҶеҶ…е®№
    testA, _ := os.Create("aa/testA")
    io.WriteString(testA, "the teatA")
    // еҰӮжһңйңҖиҰҒе®үе…ЁеҲӨж–ӯ, еҸҜд»ҘдҪҝз”Ё os.Stat й…ҚеҗҲ os.IsNotExist
    if _, err := os.Stat("aa/testA"); os.IsNotExist(err) {
        // еҪ“ж–Үд»¶дёҚеӯҳеңЁ, жүҚжү§иЎҢеҲӣе»ә
        os.Create("aa/testA")
    }
    os.Chmod("aa/testA", 0777)
}
